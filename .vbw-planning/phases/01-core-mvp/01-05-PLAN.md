```ruby
require "active_support"
require "active_support/core_ext/class/attribute"

module TrackRelay
  module Subscribers
    class Base
      class_attribute :synchronous, default: false

      def self.synchronous!
        self.synchronous = true
      end

      def deliver(payload)
        raise NotImplementedError, "#{self.class.name} must implement #deliver(payload)"
      end

      # Returns nil on success, or the StandardError on failure.
      # IMPORTANT: never re-raises inline — the Dispatcher decides loudness
      # AFTER fan-out completes. This guarantees peer subscribers always
      # receive the event, while still letting dev/test re-raise loudly.
      def handle(payload)
        if self.class.synchronous || TrackRelay.config.force_synchronous
          safe_deliver(payload)
        else
          DeliveryJob.perform_later(self.class.name, payload.to_h)
          nil
        end
      end

      # Returns nil on success, or the StandardError on failure.
      # ALWAYS logs the failure; NEVER re-raises here.
      def safe_deliver(payload)
        deliver(payload)
        nil
      rescue StandardError => e
        log_failure(e)
        e
      end

      private

      def log_failure(e)
        return unless defined?(Rails) && Rails.respond_to?(:logger)
        Rails.logger.error(
          "[track_relay] subscriber=#{self.class.name} failed: #{e.class}: #{e.message}\n" \
          "#{Array(e.backtrace).first(5).join("\n")}"
        )
      end
    end
  end
end
```

CONTRACT NOTE: `safe_deliver` and `handle` return `nil` on success or the exception on failure. They DO NOT re-raise. The Dispatcher (Task 4) collects these return values and re-raises the first one after fan-out finishes (when `swallow_subscriber_errors` is false). This is the locked decision from the reviewer fix and CONTEXT.md "re-raise in dev/test" rule.

Write `lib/track_relay/delivery_job.rb`:

```ruby
require "active_job"

module TrackRelay
  class DeliveryJob < ActiveJob::Base
    queue_as :track_relay

    def perform(subscriber_class_name, payload_hash)
      subscriber = subscriber_class_name.constantize.new
      payload = EventPayload.from_h(payload_hash)
      result = subscriber.safe_deliver(payload)
      # Async loudness — mirrors the sync Dispatcher's collect-then-reraise contract.
      # `safe_deliver` already logged via Rails.logger.error; if the host opts out of
      # swallowing, re-raise so ActiveJob's normal error path (retry / discard /
      # failed-job queue) surfaces the failure. In :test adapter, this propagates
      # synchronously through perform_now/perform_later.
      if result.is_a?(StandardError) && !TrackRelay.config.swallow_subscriber_errors
        raise result
      end
    end
  end
end
```

Write `lib/track_relay/dispatcher.rb`:
```ruby
module TrackRelay
  module Dispatcher
    NOTIFICATION = "track_relay.event"

    class << self
      def start!(notifier = ActiveSupport::Notifications)
        return @subscription if @subscription
        @subscription = notifier.subscribe(NOTIFICATION) do |*, payload|
          dispatch(payload[:event])
        end
      end

      def stop!(notifier = ActiveSupport::Notifications)
        return unless @subscription
        notifier.unsubscribe(@subscription)
        @subscription = nil
      end

      def started?
        !@subscription.nil?
      end

      private

      # Collect-then-reraise: every subscriber receives the event, then we
      # re-raise the first error (if any) when swallow_subscriber_errors is
      # false. This is the locked CONTEXT.md decision: dev/test must be loud
      # but ONE bad subscriber must NEVER prevent peers from receiving the event.
      def dispatch(payload)
        errors = []
        TrackRelay.config.subscribers.each do |subscriber|
          # Subscribers::Base#handle returns nil on success or the StandardError
          # on failure (it never re-raises inline). For non-Base subscribers
          # that ignore that contract and raise inline, we wrap defensively to
          # preserve the "peers still run" invariant.
          result =
            begin
              subscriber.handle(payload)
            rescue StandardError => e
              if defined?(Rails) && Rails.respond_to?(:logger)
                Rails.logger.error("[track_relay] non-Base subscriber #{subscriber.class} raised inline: #{e.class}: #{e.message}")
              end
              e
            end
          errors << result if result.is_a?(StandardError)
        end

        return if errors.empty?
        return if TrackRelay.config.swallow_subscriber_errors
        raise errors.first
      end
    end
  end
end
```

The defensive inline rescue is a SAFETY NET for non-Base subscribers, NOT the primary error-handling site. The contract is: `Subscribers::Base#safe_deliver` returns the exception (never raises); the dispatcher collects those returns; after fan-out it re-raises one if loudness is on. Plan 04's success criteria + this dispatcher are the two halves of the locked "re-raise in dev/test" rule.
