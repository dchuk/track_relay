Write `lib/track_relay/testing.rb`:
```ruby
module TrackRelay
  module Testing
    module_function

    def test_mode!
      return @test_subscriber if active?
      test_subscriber = Subscribers::Test.new
      @previous_subscribers = TrackRelay.config.replace_subscribers([test_subscriber])
      @test_subscriber = test_subscriber
    end

    def test_mode_off!
      return unless active?
      TrackRelay.config.replace_subscribers(@previous_subscribers)
      @previous_subscribers = nil
      @test_subscriber = nil
    end

    def active?
      !@test_subscriber.nil?
    end

    def test_subscriber
      @test_subscriber
    end
  end
end

module TrackRelay
  class << self
    def test_mode!
      Testing.test_mode!
    end

    def test_mode_off!
      Testing.test_mode_off!
    end

    def test_subscriber
      Testing.test_subscriber
    end
  end
end
```

Write `lib/track_relay/testing/minitest_assertions.rb`:
```ruby
module TrackRelay
  module Testing
    module MinitestAssertions
      def track_relay_test
        TrackRelay.test_subscriber || raise("Call TrackRelay.test_mode! before using assert_tracked")
      end

      def assert_tracked(name, **expected_params)
        events = track_relay_test.find(name)
        assert events.any?, "Expected an event :#{name} to be tracked, but found #{track_relay_test.events.map(&:name).inspect}"
        return if expected_params.empty?
        match = events.find { |e| expected_params.all? { |k, v| e.params[k] == v } }
        assert match, "Expected :#{name} with params >= #{expected_params.inspect}, got #{events.map(&:params).inspect}"
      end

      def refute_tracked(name)
        events = track_relay_test.find(name)
        refute events.any?, "Expected no event :#{name} to be tracked, but got #{events.map(&:params).inspect}"
      end
    end
  end
end
```

Write `lib/track_relay/testing/helpers.rb` as a thin wrapper that includes both Minitest assertions and (later) sets up RSpec helpers:
```ruby
require "track_relay/testing"
require "track_relay/testing/minitest_assertions"

module TrackRelay
  module Testing
    module Helpers
      include MinitestAssertions

      def self.included(base)
        base.class_eval do
          if respond_to?(:setup) && respond_to?(:teardown)
            setup { TrackRelay.test_mode! }
            teardown { TrackRelay.test_mode_off! }
          end
        end
      end
    end
  end
end
```

Note: `setup` / `teardown` blocks here only register if Minitest's hooks are available (i.e. when included in `ActiveSupport::TestCase` or `Minitest::Test`).

if defined?(RSpec)
  RSpec::Matchers.define :have_tracked do |name|
    match do |_actual|
      raise "Call TrackRelay.test_mode! before have_tracked" unless TrackRelay::Testing.active?
      events = TrackRelay.test_subscriber.find(name)
      next false if events.empty?
      next true if @expected_params.nil?
      events.any? { |e| @expected_params.all? { |k, v| e.params[k] == v } }

      @expected_params = params
    end

    failure_message do |_actual|
      seen = TrackRelay.test_subscriber.events.map { |e| {name: e.name, params: e.params} }
      msg = "expected an event :#{name} to be tracked"
      msg += " with params >= #{@expected_params.inspect}" if @expected_params
      msg + ", but got #{seen.inspect}"
    end
  end

  RSpec::Matchers.define :have_identified do |user|

Inline RSpec test:
```ruby
require "test_helper"
require "rspec/core"
require "rspec/expectations"
require "track_relay/testing"  # loads matchers when RSpec defined

class RspecMatchersTest < ActiveSupport::TestCase
  setup do
    TrackRelay.catalog do
      event :foo do
        integer :n
      end
    end
    TrackRelay.test_mode!
  end

  teardown { TrackRelay.test_mode_off! }

  include RSpec::Matchers

  test "have_tracked matches a fired event" do
    TrackRelay.track(:foo, n: 1)
    expect(self).to have_tracked(:foo)
  end

  test "have_tracked with .with(params) matches" do
    TrackRelay.track(:foo, n: 7)
    expect(self).to have_tracked(:foo).with(n: 7)
  end

  test "have_tracked fails when event missing" do
    assert_raises(RSpec::Expectations::ExpectationNotMetError) do
      expect(self).to have_tracked(:foo)
    end
  end

  test "have_tracked with wrong params fails" do
    TrackRelay.track(:foo, n: 1)
    assert_raises(RSpec::Expectations::ExpectationNotMetError) do
      expect(self).to have_tracked(:foo).with(n: 999)
    end
  end
end
```

The trick: include `RSpec::Matchers` into the test class so `expect()` works inside Minitest tests. Failures in `expect` raise `RSpec::Expectations::ExpectationNotMetError`, which we catch with `assert_raises`.
