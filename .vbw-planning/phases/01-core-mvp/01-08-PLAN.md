Write `lib/track_relay/linter.rb`:
```ruby
require "json"

module TrackRelay
  class Linter
    Report = Struct.new(:event_name, :signatures, :total, keyword_init: true)
    Signature = Struct.new(:params, :count, keyword_init: true)

    attr_reader :malformed_lines

    def initialize(jsonl_path)
      @jsonl_path = jsonl_path
      @malformed_lines = 0
    end

    def report
      groups = Hash.new { |h, k| h[k] = Hash.new(0) }
      read_lines do |entry|
        event = entry["event"]
        signature = Array(entry["params"]).sort
        groups[event][signature] += 1
      end

      groups.map { |event, signatures|
        sig_list = signatures.map { |params, count| Signature.new(params: params, count: count) }
        total = sig_list.sum(&:count)
        Report.new(event_name: event, signatures: sig_list.sort_by { |s| -s.count }, total: total)
      }.sort_by { |r| -r.total }
    end

    def print(io = $stdout)
      reports = report
      io.puts "# track_relay untyped event audit"
      io.puts "# source: #{@jsonl_path}"
      io.puts "# events: #{reports.size}; total occurrences: #{reports.sum(&:total)}"
      io.puts ""
      reports.each do |r|
        io.puts "event :#{r.event_name}  (#{r.total} total)"
        r.signatures.each do |sig|
          io.puts "  - params=[#{sig.params.join(", ")}]  count=#{sig.count}"
        end
        io.puts ""
      end
      io.puts "# #{@malformed_lines} malformed line(s) skipped" if @malformed_lines.positive?
    end

    def to_json
      JSON.generate(report.map { |r|
        {
          event: r.event_name,
          total: r.total,
          signatures: r.signatures.map { |s| {params: s.params, count: s.count} }
        }
      })
    end

    private

    def read_lines
      return unless File.exist?(@jsonl_path.to_s)
      File.foreach(@jsonl_path.to_s) do |line|
        line = line.strip
        next if line.empty?
        begin
          yield JSON.parse(line)
        rescue JSON::ParserError
          @malformed_lines += 1
        end
      end
    end
  end
end
```

Write `lib/tasks/track_relay.rake`:
```ruby
namespace :track_relay do
  desc "Audit untyped events captured in the JSONL sink (config.untyped_log_path)"
  task lint: :environment do
    path = TrackRelay.config.untyped_log_path
    if path.nil?
      abort <<~MSG
        track_relay: untyped_log_path is not configured.
        Set it in config/initializers/track_relay.rb:

          TrackRelay.configure do |c|
            c.untyped_log_path = Rails.root.join("tmp/track_relay_untyped.jsonl")
            c.subscribe TrackRelay::Subscribers::Logger.new
          end
      MSG
    end

    linter = TrackRelay::Linter.new(path)
    linter.print
  end

  desc "Audit untyped events and emit JSON report to stdout"
  task "lint:json" => :environment do
    path = TrackRelay.config.untyped_log_path
    abort "track_relay: untyped_log_path is not configured" if path.nil?
    linter = TrackRelay::Linter.new(path)
    puts linter.to_json
  end
end
```

```ruby
require "test_helper"

class LinterRakeTaskTest < ActiveSupport::TestCase
  setup do
    Rake.application = Rake::Application.new
    Rake.application.rake_require("track_relay", [File.expand_path("../../../lib/tasks", __dir__)])
    Rake::Task.define_task(:environment)  # stub, since :environment depends on full Rails app load
  end

  test "rake track_relay:lint aborts NONZERO when untyped_log_path is unset" do
    TrackRelay.config.untyped_log_path = nil
    err = assert_raises(SystemExit) do
      capture_io { Rake::Task["track_relay:lint"].invoke }
    end
    refute_equal 0, err.status, "Expected nonzero exit status when untyped_log_path is unset"
  end

  test "rake track_relay:lint prints report when path is set and file exists" do
    Tempfile.create("untyped.jsonl") do |f|
      # Use the canonical Logger JSONL shape (Plan 05 contract).
      f.puts({event: "ad_hoc", params: %w[foo bar], controller: "ArticlesController", action: "show", timestamp: Time.now.iso8601}.to_json)
      f.flush
      TrackRelay.config.untyped_log_path = f.path
      Rake::Task["track_relay:lint"].reenable
      out, _err = capture_io { Rake::Task["track_relay:lint"].invoke }
      assert_match(/event :ad_hoc/, out)
      assert_match(/params=\[bar, foo\]/, out)
    end
  end

  test "rake track_relay:lint:json emits JSON" do
    Tempfile.create("untyped.jsonl") do |f|

      Rake::Task["track_relay:lint:json"].reenable

  end
end
```
