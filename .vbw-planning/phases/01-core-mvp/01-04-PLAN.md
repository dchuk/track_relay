Write `lib/track_relay/instrumenter.rb` as the implementation behind `TrackRelay.track` / `.identify`. Module method shape:

```ruby
require "active_support/notifications"

module TrackRelay
  module Instrumenter
    NOTIFICATION = "track_relay.event"
    IDENTIFY_NOTIFICATION = "track_relay.identify"

    module_function

    def track(name, **params)
      reserved, event_params = partition_reserved(params)
      Current.set(**reserved) do
        definition = Catalog.lookup(name)
        payload = build_payload(name: name, definition: definition, params: event_params)
        return unless payload  # skipped untyped + disallowed case
        validate(payload)
        ActiveSupport::Notifications.instrument(NOTIFICATION, event: payload)
      end
    end

    def identify(user, **user_properties)
      ActiveSupport::Notifications.instrument(
        IDENTIFY_NOTIFICATION,
        user: user,
        properties: user_properties
      )
    end

    def partition_reserved(params)
      reserved = {}
      remaining = {}
      params.each do |k, v|
        if RESERVED_KEYS.include?(k)
          reserved[k] = v
        else
          remaining[k] = v
        end
      end
      [reserved, remaining]
    end

    def build_payload(name:, definition:, params:)
      context = current_context
      if definition
        EventPayload.new(definition: definition, params: params, context: context)
      elsif TrackRelay.config.untyped_events_allowed
        EventPayload.untyped(name: name, params: params, context: context)
      else
        raise UnknownEventError, "Unknown event :#{name}; declare it in your catalog or set config.untyped_events_allowed = true"
      end
    end

    def current_context
      {
        user: Current.user,
        controller: Current.controller&.class&.name,
        action: Current.controller&.respond_to?(:action_name) ? Current.controller.action_name : nil,
        client_id: Current.client_id,
        visit: Current.visit,
        request_id: Current.request&.request_id
      }
    end

    def validate(payload)
      payload.validate!
    rescue ValidationError => e
      raise if TrackRelay.config.raise_on_validation_error
      Rails.logger.error("[track_relay] validation failed for :#{payload.name}: #{e.message}") if defined?(Rails)
      throw :track_relay_skip
    end
  end
end
```

Refactor: instead of `throw :track_relay_skip` use a regular early-return. Adjust shape so `track` checks the return of `validate` and skips instrument on swallow path. Use whatever idiom is cleanest in StandardRB.

Wire into `lib/track_relay.rb`:
```ruby
require "track_relay/instrumenter"

module TrackRelay
  def self.track(name, **params)
    Instrumenter.track(name, **params)
  end

  def self.identify(user, **user_properties)
    Instrumenter.identify(user, **user_properties)
  end
end
```

CRITICAL routing rule (per frontmatter must_have): `:user`, `:request`, `:client_id` are passed to `Current.set(**current_attrs) { ... }` (these ARE Current attributes). `:visitor_token` is NOT a Current attribute (Current carries `:visit`, an Ahoy-style record, not a raw token). `:visitor_token` is extracted from params (so it never appears in `payload.params`) and merged directly into `payload.context[:visitor_token]` when the payload is built. Implement this by changing `partition_reserved` to return three buckets: `current_attrs` (for `Current.set`), `direct_context` (for `:visitor_token`), and `event_params` (the rest). Then in `track`:

```ruby
def track(name, **params)
  current_attrs, direct_context, event_params = partition_reserved(params)
  Current.set(**current_attrs) do
    definition = Catalog.lookup(name)
    payload = build_payload(name: name, definition: definition, params: event_params, extra_context: direct_context)
    return unless payload
    validate(payload)
    ActiveSupport::Notifications.instrument(NOTIFICATION, event: payload)
  end
end
```

`build_payload` merges `extra_context` into `current_context` so `payload.context[:visitor_token]` is set. Add a code comment referencing this plan's must-have for the rationale.

Write `test/integration/track_test.rb` using `ActiveSupport::Notifications.subscribed { }` (per RESEARCH.md section 2 example):

```ruby
require "test_helper"

class TrackTest < ActiveSupport::TestCase
  setup do
    TrackRelay.catalog do
      event :article_viewed do
        integer :article_id, required: true
        string :slug, required: true
        string :category
      end
    end
  end

  test "track instruments track_relay.event with EventPayload" do
    events = []
    ActiveSupport::Notifications.subscribed(->(e) { events << e }, "track_relay.event") do
      TrackRelay.track(:article_viewed, article_id: 1, slug: "hello", category: "news")
    end
    assert_equal 1, events.size
    payload = events.first.payload[:event]
    assert_equal :article_viewed, payload.name
    assert_equal({article_id: 1, slug: "hello", category: "news"}, payload.params)
  end

  test "track coerces and validates" do
    events = []
    ActiveSupport::Notifications.subscribed(->(e) { events << e }, "track_relay.event") do
      TrackRelay.track(:article_viewed, article_id: "42", slug: "x")  # string coerced to int
    end
    assert_equal 42, events.first.payload[:event].params[:article_id]
  end

  test "track raises ValidationError when raise_on_validation_error is true" do
    TrackRelay.config.raise_on_validation_error = true
    assert_raises(TrackRelay::ValidationError) do
      TrackRelay.track(:article_viewed, slug: "x")  # missing required :article_id
    end
  end

  test "track swallows ValidationError when raise_on_validation_error is false" do
    TrackRelay.config.raise_on_validation_error = false
    events = []
    ActiveSupport::Notifications.subscribed(->(e) { events << e }, "track_relay.event") do
      TrackRelay.track(:article_viewed, slug: "x")  # missing required
    end
    assert_empty events  # nothing instrumented when validation fails + swallow
  end

  test "reserved keys extracted into Current and context, never params" do
    user = Object.new
    events = []
    ActiveSupport::Notifications.subscribed(->(e) { events << e }, "track_relay.event") do
      TrackRelay.track(:article_viewed, article_id: 1, slug: "x", user: user)
    end
    payload = events.first.payload[:event]
    refute_includes payload.params.keys, :user
    assert_equal user, payload.context[:user]
  end

  test "visitor_token is stored on context, never on params, never on Current" do
    events = []
    ActiveSupport::Notifications.subscribed(->(e) { events << e }, "track_relay.event") do
      TrackRelay.track(:article_viewed, article_id: 1, slug: "x", visitor_token: "vt-123")
    end
    payload = events.first.payload[:event]
    refute_includes payload.params.keys, :visitor_token
    assert_equal "vt-123", payload.context[:visitor_token]
  end
end
```
