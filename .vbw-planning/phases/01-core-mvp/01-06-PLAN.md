Write `lib/track_relay/railtie.rb`:
```ruby
require "rails/railtie"

module TrackRelay
  class Railtie < Rails::Railtie
    initializer "track_relay.catalog_autoload" do |app|
      catalog_dir = app.root.join("config", "track_relay")

      if catalog_dir.exist?
        Rails.autoloaders.main.ignore(catalog_dir)
      end

      app.config.to_prepare do
        Dir.glob("#{catalog_dir}/**/*.rb").sort.each do |file|
          load file
        end
      end
    end

    initializer "track_relay.start_dispatcher", after: :load_config_initializers do
      ActiveSupport.on_load(:after_initialize) do
        TrackRelay::Dispatcher.start!
      end
    end
  end
end
```

Note: `ActiveSupport.on_load(:after_initialize)` may not be a real hook name (`after_initialize` is fired via `config.after_initialize`). The robust pattern is:
```ruby
initializer "track_relay.start_dispatcher" do |app|
  app.config.after_initialize do
    TrackRelay::Dispatcher.start!
  end
end
```

Write `lib/track_relay/controller_tracking.rb`:
```ruby
require "active_support/concern"

module TrackRelay
  module ControllerTracking
    extend ActiveSupport::Concern

    included do
      before_action :_track_relay_set_current
    end

    def track(name, **params)
      TrackRelay.track(name, **params)
    end

    private

    def _track_relay_set_current
      TrackRelay::Current.controller = self
      TrackRelay::Current.request = request
      TrackRelay::Current.client_id = _track_relay_client_id_from_cookie
    end

    def _track_relay_client_id_from_cookie
      ga_cookie = cookies["_ga"]
      return nil if ga_cookie.nil? || ga_cookie.empty?
      # _ga cookie format: "GA1.2.123456789.1234567890" — last two segments form the client_id
      parts = ga_cookie.split(".")
      return nil if parts.size < 4
      "#{parts[-2]}.#{parts[-1]}"
    end
  end
end
```

Write `test/internal/app/jobs/welcome_email_job.rb`:
```ruby
class WelcomeEmailJob < ActiveJob::Base
  include TrackRelay::JobTracking

  def perform(user_id, visitor_token)
    user = OpenStruct.new(id: user_id, last_visitor_token: visitor_token)
    TrackRelay::Current.set(user: user) do
      track :welcome_email_sent, user_id: user_id, visitor_token: visitor_token, template: "v3"
    end
  end
end
```

Note `:user_id` is a regular param (not reserved); `:visitor_token` is a reserved key extracted into context. The `Current.set(user: user)` block runs before track to populate Current — which means `Current.user = user` at instrument time → context[:user] = user.

```ruby
require "test_helper"
require "ostruct"

class JobTrackingTest < ActiveSupport::TestCase
  include ActiveJob::TestHelper

  setup do
    TrackRelay.catalog do
      event :welcome_email_sent do
        integer :user_id, required: true
        string :template, required: true
      end
    end
  end

  test "track helper inside a job fires an event with Current populated" do
    captured = nil
    sub = ActiveSupport::Notifications.subscribe("track_relay.event") do |*, payload|
      captured = payload[:event]
    end
    perform_enqueued_jobs do
      WelcomeEmailJob.perform_later(7, "vt-abc")
    end
    refute_nil captured
    assert_equal :welcome_email_sent, captured.name
    assert_equal 7, captured.params[:user_id]
    assert_equal "v3", captured.params[:template]
    assert_equal "vt-abc", captured.context[:visitor_token]  # reserved-key extraction
    assert captured.context[:user], "Current.user should be populated via Current.set block"
  ensure
    ActiveSupport::Notifications.unsubscribe(sub) if sub
  end

  test "Current is reset before job perform (Executor contract)" do
    # Set Current outside the job, perform the job, assert it was nil at job start
    TrackRelay::Current.user = "request_user"
    saw_at_start = nil
    sub = ActiveSupport::Notifications.subscribe("track_relay.event") do |*, payload|
      saw_at_start = payload[:event].context[:user]
    end
    # Job perform: pre-set-block, Current.user should be nil
    job_class = Class.new(ActiveJob::Base) do
      include TrackRelay::JobTracking
      def perform
        # do NOT use Current.set; observe Current.user is nil
        track :welcome_email_sent, user_id: 1, template: "vX"
      end
    end
    Object.const_set(:_TempProbeJob, job_class)
    begin
      perform_enqueued_jobs do
        _TempProbeJob.perform_later
      end
      assert_nil saw_at_start, "Executor should reset Current before job perform"
    ensure
      Object.send(:remove_const, :_TempProbeJob)
      ActiveSupport::Notifications.unsubscribe(sub) if sub
    end
  end
end
```

The second test documents the Executor-reset gotcha from RESEARCH.md section 5.
